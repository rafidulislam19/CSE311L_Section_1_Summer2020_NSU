1.SELECT Round (max(salary)) "Maximum", Round (sum(salary)) "sum", Round(avg(salary)) "average" from employees;

2.SELECT job_id;
Round(Max (salary)) "Maximum",
Round(Min (salary)) "Minimum",
Round(sum (salary)) "sum",
Round(avg (salary)) "avg",
from employees
group by job_id;

3.SELECT job_id, COUNT(*) FROM employees GROUP BY job_id;

4.SELECT manager_id, MIN(salary)
FROM employees
WHERE manager_id IS NOT NULL
GROUP BY manager_id
ORDER BY MIN(salary) DESC;
